	// drop table if exists employeedepartments;

	// drop table if exists departments;

	CREATE TABLE IF NOT EXISTS Departments(
		Id INT PRIMARY KEY,
		DepartmentName VARCHAR(100)
	);


	CREATE TABLE IF NOT EXISTS Employees(
		Id				INT PRIMARY KEY,
		Name			VARCHAR(100),
		DepartmentId	INT,
		Salary			INT,
		FOREIGN KEY(DepartmentId) REFERENCES Departments(Id)
	);


	CREATE TABLE IF NOT EXISTS EmployeeDepartments(
		EmployeeId INT,
		DepartmentId INT,
		PRIMARY KEY (EmployeeId, DepartmentId),
		FOREIGN KEY (EmployeeId) REFERENCES Employees(Id),
		FOREIGN KEY (DepartmentId) REFERENCES Departments(Id)
	);

Задачи

	Найти сотрудников с зарплатой выше средней

		select name, salary
		from employees
		where salary > (select avg(salary) from employees);


	Найти количество сотрудников в каждом отделе

		select d.departmentname, count(e.name)
		from departments d
		join employees e
		on d.id = e.departmentid
		group by d.departmentname;


	Найти второго по зарплате сотрудника

		select max(salary)
		from employees
		where salary< (select max(salary) from employees);


	Получить список уникальных зарплат

		select distinct salary
		from employees
		order by salary;


	Найти сотрудников с одинаковой зарплатой

		select name, salary
		from employees
		where salary in
		(
			select salary from employees
			group by salary
			having count(*) > 1
		);


	Найти сотрудников, которые работают в нескольких отделах

		Первый вариант решения

			select distinct Id, Name
			from employees e
			join employeedepartments ed on e.id = ed.EmployeeId
			where EmployeeId in
			(
				select EmployeeId
				from EmployeeDepartments
				group by EmployeeId
				having count(*) > 1
			);


		Второй вариант решения

			select e.id, e.name
			from employees e
			join employeedepartments ed
			on e.Id = ed.EmployeeId
			group by e.Id, e.name
			having count(ed.departmentid) > 1;


	Найти среднюю зарплату в каждом отделе

		select DepartmentId, avg(Salary)
		from employees
		group by DepartmentId;


	Получить топ-3 сотрудников по зарплате

		select Name, Salary
		from employees
		order by Salary desc
		limit 3;


	Получить имена сотрудников с максимальной зарплатой в каждом отделе

		select e.Name, e.Salary, d.DepartmentName
		from Employees e
		join Departments d
		on e.DepartmentId = d.Id
		where e.Salary =
		(
			SELECT max(Salary)
			from Employees
			where DepartmentId = d.Id
		);
